  static float clamp01(float x){ return x<0?0:x>1?1:x; }
  static float smoothstep(float t){ t=clamp01(t); return t*t*(3-2*t); }
  static float fadeIn(float t,float s,float e,float d=2.0f){
      return smoothstep(clamp01((t-s)/d));
  }
  static float fadeOut(float t,float s,float e,float d=2.0f){
      return 1.0f-smoothstep(clamp01((t-(e-d))/d));
  }
  static float sceneFade(float t,float s,float e,float d=2.0f){
      return clamp01(fadeIn(t,s,e,d)*fadeOut(t,s,e,d));
  }
  static float pulse(float t,float freq=1.0f){
      return 0.5f+0.5f*sinf(t*freq*6.2832f);
  }
